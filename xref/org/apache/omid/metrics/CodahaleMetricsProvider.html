49      public static final Pattern CODAHALE_METRICS_CONFIG_PATTERN = Pattern
50              .compile("(csv|slf4j|console|graphite):(.+):(\\d+):(DAYS|HOURS|MICROSECONDS|MILLISECONDS|MINUTES|NANOSECONDS|SECONDS)");
51  
52      private MetricRegistry metrics = new MetricRegistry();
53      private List<ScheduledReporter> reporters = new ArrayList<>();
54  
55      private final int metricsOutputFrequencyInSecs;
56  
57      public static CodahaleMetricsProvider createCodahaleMetricsProvider(
58              List<String> metricsConfigs) throws IOException {
59  
60          CodahaleMetricsConfig codahaleConfig = new CodahaleMetricsConfig();
61  
62          for (String metricConfig : metricsConfigs) {
63              Matcher matcher = CODAHALE_METRICS_CONFIG_PATTERN.matcher(metricConfig);
64              if (matcher.matches()) {
65  
66                  String reporter = matcher.group(1);
67                  String reporterConfig = matcher.group(2);
68                  codahaleConfig.setOutputFreqInSecs(Integer.valueOf(matcher.group(3)));
69  
70                  switch (reporter) {
71                      case "csv":
72                          codahaleConfig.addReporter(Reporter.CSV);
73                          codahaleConfig.setCsvDir(reporterConfig);
74                          break;
75                      case "slf4j":
76                          codahaleConfig.addReporter(Reporter.SLF4J);
77                          codahaleConfig.setSlf4jLogger(reporterConfig);
78                          break;
79                      case "graphite":
80                          codahaleConfig.addReporter(Reporter.GRAPHITE);
81                          codahaleConfig.setGraphiteHostConfig(reporterConfig);
82                          break;
83                      case "console":
84                          codahaleConfig.addReporter(Reporter.CONSOLE);
85                          break;
86                      default:
87                          LOG.warn("Reporter {} unknown", reporter);
88                          break;
89                  }
90              } else {
91                  LOG.error("Pattern {} not recognized", metricConfig);
92              }
93          }
94          CodahaleMetricsProvider provider = new CodahaleMetricsProvider(codahaleConfig);
95          provider.startMetrics();
96          return provider;
97      }

99      public CodahaleMetricsProvider(CodahaleMetricsConfig conf) throws IOException {
100         metricsOutputFrequencyInSecs = conf.getOutputFreqInSecs();
101         int reporterCount = 0;
102         for (Reporter reporter : conf.getReporters()) {
103             ScheduledReporter codahaleReporter = null;
104             switch (reporter) {
105                 case CONSOLE:
106                     codahaleReporter = createAndGetConfiguredConsoleReporter();
107                     break;
108                 case GRAPHITE:
109                     codahaleReporter = createAndGetConfiguredGraphiteReporter(conf.getPrefix(),
110                                                                               conf.getGraphiteHostConfig());
111                     break;
112                 case CSV:
113                     codahaleReporter = createAndGetConfiguredCSVReporter(conf.getPrefix(),
114                                                                          conf.getCsvDir());
115                     break;
116                 case SLF4J:
117                     codahaleReporter = createAndGetConfiguredSlf4jReporter(conf.getSlf4jLogger());
118                     break;
119             }
120             if (codahaleReporter != null) {
121                 reporters.add(codahaleReporter);
122                 reporterCount++;
123             }
124         }
125         if (reporterCount == 0) {
126             LOG.warn("No metric reporters found, so metrics won't be available");
127         }
128         startMetrics();
129     }

131     @Override
132     public void startMetrics() {
133         for (ScheduledReporter r : reporters) {
134             LOG.info("Starting metrics reporter {} reporting every {} Secs",
135                      r.getClass().getCanonicalName(), metricsOutputFrequencyInSecs);
136             r.start(metricsOutputFrequencyInSecs, TimeUnit.SECONDS);
137         }
138     }
139 
140     @Override
141     public void stopMetrics() {
142         for (ScheduledReporter r : reporters) {
143             r.report();
144             LOG.info("Stopping reporter {}", r.toString());
145             r.stop();
146         }
147     }

179     private ScheduledReporter createAndGetConfiguredConsoleReporter() {
180         return ConsoleReporter.forRegistry(metrics)
181                 .convertRatesTo(TimeUnit.SECONDS)
182                 .convertDurationsTo(TimeUnit.MILLISECONDS)
183                 .build();
184     }
185 
186     private ScheduledReporter createAndGetConfiguredGraphiteReporter(String prefix, String graphiteHost) {
187         LOG.info("Configuring Graphite reporter. Sendig data to host:port {}", graphiteHost);
188         HostAndPort addr = HostAndPort.fromString(graphiteHost);
189 
190         final Graphite graphite = new Graphite(
191                 new InetSocketAddress(addr.getHostText(), addr.getPort()));
192 
193         return GraphiteReporter.forRegistry(metrics)
194                 .prefixedWith(prefix)
195                 .convertRatesTo(TimeUnit.SECONDS)
196                 .convertDurationsTo(TimeUnit.MILLISECONDS)
197                 .filter(MetricFilter.ALL)
198                 .build(graphite);
199     }

201     private ScheduledReporter createAndGetConfiguredCSVReporter(String prefix, String csvDir) throws IOException {
202         // NOTE:
203         // 1) metrics output files are exclusive to a given process
204         // 2) the output directory must exist
205         // 3) if output files already exist they are not overwritten and there is no metrics output
206         File outputDir;
207         if (Strings.isNullOrEmpty(prefix)) {
208             outputDir = new File(csvDir, prefix);
209         } else {
210             outputDir = new File(csvDir);
211         }
212         FileUtils.forceMkdir(outputDir);
213         LOG.info("Configuring stats with csv output to directory [{}]", outputDir.getAbsolutePath());
214         return CsvReporter.forRegistry(metrics)
215                 .convertRatesTo(TimeUnit.SECONDS)
216                 .convertDurationsTo(TimeUnit.MILLISECONDS)
217                 .build(outputDir);
218     }

221     private ScheduledReporter createAndGetConfiguredSlf4jReporter(String slf4jLogger) {
222         LOG.info("Configuring stats with SLF4J with logger {}", slf4jLogger);
223         return Slf4jReporter.forRegistry(metrics)
224                 .outputTo(LoggerFactory.getLogger(slf4jLogger))
225                 .convertRatesTo(TimeUnit.SECONDS)
226                 .convertDurationsTo(TimeUnit.MILLISECONDS)
227                 .build();
228     }
