107         this.leaseManager = leaseManager;

118         batchSizeHistogram = metrics.histogram(name("tso", "persist", "batchsize"));
119         timeoutMeter = metrics.meter(name("tso", "persist", "timeout"));
120 
121         // FIXME consider putting something more like a phased strategy here to avoid
122         // all the syscalls
123         final TimeoutBlockingWaitStrategy timeoutStrategy
124                 = new TimeoutBlockingWaitStrategy(config.getBatchPersistTimeoutInMs(), TimeUnit.MILLISECONDS);
125 
126         persistRing = RingBuffer.createSingleProducer(
127                 PersistEvent.EVENT_FACTORY, 1 << 20, timeoutStrategy); // 2^20 entries in ringbuffer
128         SequenceBarrier persistSequenceBarrier = persistRing.newBarrier();
129         BatchEventProcessor<PersistEvent> persistProcessor = new BatchEventProcessor<>(
130                 persistRing,
131                 persistSequenceBarrier,
132                 this);
133         persistRing.addGatingSequences(persistProcessor.getSequence());
134         persistProcessor.setExceptionHandler(new FatalExceptionHandler(panicker));
135 
136         ExecutorService persistExec = Executors.newSingleThreadExecutor(
137                 new ThreadFactoryBuilder().setNameFormat("persist-%d").build());
138         persistExec.submit(persistProcessor);

142     @Override
143     public void onEvent(PersistEvent event, long sequence, boolean endOfBatch) throws Exception {
144 
145         switch (event.getType()) {
146             case COMMIT:
147                 event.getMonCtx().timerStart("commitPersistProcessor");
148                 // TODO: What happens when the IOException is thrown?
149                 writer.addCommittedTransaction(event.getStartTimestamp(), event.getCommitTimestamp());
150                 batch.addCommit(event.getStartTimestamp(), event.getCommitTimestamp(), event.getChannel(),
151                                 event.getMonCtx());
152                 break;
153             case ABORT:
154                 sendAbortOrIdentifyFalsePositive(event.getStartTimestamp(), event.isRetry(), event.getChannel(),
155                                                  event.getMonCtx());
156                 break;
157             case TIMESTAMP:
158                 event.getMonCtx().timerStart("timestampPersistProcessor");
159                 batch.addTimestamp(event.getStartTimestamp(), event.getChannel(), event.getMonCtx());
160                 break;
161         }
162         if (batch.isFull() || endOfBatch) {
163             maybeFlushBatch();
164         }
165 
166     }

168     private void sendAbortOrIdentifyFalsePositive(long startTimestamp, boolean isRetry, Channel channel,
169                                                   MonitoringContext monCtx) {
170 
171         if (!isRetry) {
172             reply.abortResponse(startTimestamp, channel, monCtx);
173             return;
174         }
175 
176         // If is a retry, we must check if it is a already committed request abort.
177         // This can happen because a client could have missed the reply, so it
178         // retried the request after a timeout. So we added to the batch and when
179         // it's flushed we'll add events to the retry processor in order to check
180         // for false positive aborts. It needs to be done after the flush in case
181         // the commit has occurred but it hasn't been persisted yet.
182         batch.addUndecidedRetriedRequest(startTimestamp, channel, monCtx);
183     }

191     /**
192      * Flush the current batch if it's full, or the timeout has been elapsed since the last flush.
193      */
194     private void maybeFlushBatch() {
195         if (batch.isFull()) {
196             flush();
197         } else if ((System.nanoTime() - lastFlush) > TimeUnit.MILLISECONDS.toNanos(batchPersistTimeoutInMs)) {
198             timeoutMeter.mark();
199             flush();
200         }
201     }
202 
203     @VisibleForTesting
204     synchronized void flush() {
205         lastFlush = System.nanoTime();
206 
207         boolean areWeStillMaster = true;
208         if (!leaseManager.stillInLeasePeriod()) {
209             // The master TSO replica has changed, so we must inform the
210             // clients about it when sending the replies and avoid flushing
211             // the current batch of TXs
212             areWeStillMaster = false;
213             // We need also to clear the data in the buffer
214             writer.clearWriteBuffer();
215             LOG.trace("Replica {} lost mastership before flushing data", tsoHostAndPort);
216         } else {
217             try {
218                 writer.flush();
219             } catch (IOException e) {
220                 panicker.panic("Error persisting commit batch", e.getCause());
221             }
222             batchSizeHistogram.update(batch.getNumEvents());
223             if (!leaseManager.stillInLeasePeriod()) {
224                 // If after flushing this TSO server is not the master
225                 // replica we need inform the client about it
226                 areWeStillMaster = false;
227                 LOG.warn("Replica {} lost mastership after flushing data", tsoHostAndPort);
228             }
229         }
230         flushTimer.update((System.nanoTime() - lastFlush));
231         batch.sendRepliesAndReset(reply, retryProc, areWeStillMaster);
232 
233     }

268     public static class Batch {
269 
270         final PersistEvent[] events;
271         final int maxBatchSize;
272         int numEvents;
273 
274         Batch(int maxBatchSize) {
275             assert (maxBatchSize > 0);
276             this.maxBatchSize = maxBatchSize;
277             events = new PersistEvent[maxBatchSize];
278             numEvents = 0;
279             for (int i = 0; i < maxBatchSize; i++) {
280                 events[i] = new PersistEvent();
281             }
282         }
283 
284         boolean isFull() {
285             assert (numEvents <= maxBatchSize);
286             return numEvents == maxBatchSize;
287         }
288 
289         int getNumEvents() {
290             return numEvents;
291         }
292 
293         void addCommit(long startTimestamp, long commitTimestamp, Channel c, MonitoringContext monCtx) {
294             if (isFull()) {
295                 throw new IllegalStateException("batch full");
296             }
297             int index = numEvents++;
298             PersistEvent e = events[index];
299             PersistEvent.makePersistCommit(e, startTimestamp, commitTimestamp, c, monCtx);
300         }
301 
302         void addUndecidedRetriedRequest(long startTimestamp, Channel c, MonitoringContext monCtx) {
303             if (isFull()) {
304                 throw new IllegalStateException("batch full");
305             }
306             int index = numEvents++;
307             PersistEvent e = events[index];
308             // We mark the event as an ABORT retry to identify the events to send
309             // to the retry processor
310             PersistEvent.makePersistAbort(e, startTimestamp, true, c, monCtx);
311         }
312 
313         void addTimestamp(long startTimestamp, Channel c, MonitoringContext monCtx) {
314             if (isFull()) {
315                 throw new IllegalStateException("batch full");
316             }
317             int index = numEvents++;
318             PersistEvent e = events[index];
319             PersistEvent.makePersistTimestamp(e, startTimestamp, c, monCtx);
320         }
321 
322         void sendRepliesAndReset(ReplyProcessor reply, RetryProcessor retryProc, boolean isTSOInstanceMaster) {
323             for (int i = 0; i < numEvents; i++) {
324                 PersistEvent e = events[i];
325                 switch (e.getType()) {
326                     case TIMESTAMP:
327                         e.getMonCtx().timerStop("timestampPersistProcessor");
328                         reply.timestampResponse(e.getStartTimestamp(), e.getChannel(), e.getMonCtx());
329                         break;
330                     case COMMIT:
331                         e.getMonCtx().timerStop("commitPersistProcessor");
332                         if (isTSOInstanceMaster) {
333                             reply.commitResponse(false, e.getStartTimestamp(), e.getCommitTimestamp(), e.getChannel(),
334                                                  e.getMonCtx());
335                         } else {
336                             // The client will need to perform heuristic actions to determine the output
337                             reply.commitResponse(true, e.getStartTimestamp(), e.getCommitTimestamp(), e.getChannel(),
338                                                  e.getMonCtx());
339                         }
340                         break;
341                     case ABORT:
342                         if (e.isRetry()) {
343                             retryProc.disambiguateRetryRequestHeuristically(e.getStartTimestamp(), e.getChannel(),
344                                                                             e.getMonCtx());
345                         } else {
346                             LOG.error("We should not be receiving non-retried aborted requests in here");
347                         }
348                         break;
349                     default:
350                         LOG.error("We should receive only COMMIT or ABORT event types. Received {}", e.getType());
351                         break;
352                 }
353             }
354             numEvents = 0;
355         }
356 
357     }

359     public final static class PersistEvent {
360 
361         private MonitoringContext monCtx;
362 
363         enum Type {
364             TIMESTAMP, COMMIT, ABORT
365         }
366 
367         private Type type = null;
368         private Channel channel = null;
369 
370         private boolean isRetry = false;
371         private long startTimestamp = 0;
372         private long commitTimestamp = 0;
373 
374         static void makePersistCommit(PersistEvent e, long startTimestamp, long commitTimestamp, Channel c,
375                                       MonitoringContext monCtx) {
376             e.type = Type.COMMIT;
377             e.startTimestamp = startTimestamp;
378             e.commitTimestamp = commitTimestamp;
379             e.channel = c;
380             e.monCtx = monCtx;
381         }
382 
383         static void makePersistAbort(PersistEvent e, long startTimestamp, boolean isRetry, Channel c,
384                                      MonitoringContext monCtx) {
385             e.type = Type.ABORT;
386             e.startTimestamp = startTimestamp;
387             e.isRetry = isRetry;
388             e.channel = c;
389             e.monCtx = monCtx;
390         }
391 
392         static void makePersistTimestamp(PersistEvent e, long startTimestamp, Channel c, MonitoringContext monCtx) {
393             e.type = Type.TIMESTAMP;
394             e.startTimestamp = startTimestamp;
395             e.channel = c;
396             e.monCtx = monCtx;
397         }

400             return monCtx;
401         }

403         Type getType() {
