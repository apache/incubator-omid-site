56      @Test(timeOut = 30_000)
57      public void testIsCommitted(ITestContext context) throws Exception {
58          TransactionManager tm = newTransactionManager(context);
59          TTable table = new TTable(hbaseConf, TEST_TABLE);
60  
61          HBaseTransaction t1 = (HBaseTransaction) tm.begin();
62  
63          Put put = new Put(row1);
64          put.add(family, qualifier, data1);
65          table.put(t1, put);
66          tm.commit(t1);
67  
68          HBaseTransaction t2 = (HBaseTransaction) tm.begin();
69          put = new Put(row2);
70          put.add(family, qualifier, data1);
71          table.put(t2, put);
72          table.getHTable().flushCommits();
73  
74          HBaseTransaction t3 = (HBaseTransaction) tm.begin();
75          put = new Put(row2);
76          put.add(family, qualifier, data1);
77          table.put(t3, put);
78          tm.commit(t3);
79  
80          HTable htable = new HTable(hbaseConf, TEST_TABLE);
81          HBaseCellId hBaseCellId1 = new HBaseCellId(htable, row1, family, qualifier, t1.getStartTimestamp());
82          HBaseCellId hBaseCellId2 = new HBaseCellId(htable, row2, family, qualifier, t2.getStartTimestamp());
83          HBaseCellId hBaseCellId3 = new HBaseCellId(htable, row2, family, qualifier, t3.getStartTimestamp());
84  
85          HBaseTransactionClient hbaseTm = (HBaseTransactionClient) newTransactionManager(context);
86          assertTrue("row1 should be committed", hbaseTm.isCommitted(hBaseCellId1));
87          assertFalse("row2 should not be committed for kv2", hbaseTm.isCommitted(hBaseCellId2));
88          assertTrue("row2 should be committed for kv3", hbaseTm.isCommitted(hBaseCellId3));
89      }

133     @Test(timeOut = 30_000)
134     public void testReadCommitTimestampFromCommitTable(ITestContext context) throws Exception {
135 
136         final long NON_EXISTING_CELL_TS = 1000L;
137 
138         PostCommitActions syncPostCommitter =
139                 spy(new HBaseSyncPostCommitter(new NullMetricsProvider(), getCommitTable(context).getClient()));
140         AbstractTransactionManager tm = (AbstractTransactionManager) newTransactionManager(context, syncPostCommitter);
141         // The following line emulates a crash after commit that is observed in (*) below
142         doThrow(new RuntimeException()).when(syncPostCommitter).updateShadowCells(any(HBaseTransaction.class));
143 
144         // Test that a non-existing cell timestamp returns an empty result
145         Optional<CommitTimestamp> optionalCT = tm.commitTableClient.getCommitTimestamp(NON_EXISTING_CELL_TS).get();
146         assertFalse(optionalCT.isPresent());
147 
148         try (TTable table = new TTable(hbaseConf, TEST_TABLE)) {
149             // Test that we get an invalidation mark for an invalidated transaction
150 
151             // Start a transaction and invalidate it before commiting it
152             HBaseTransaction tx1 = (HBaseTransaction) tm.begin();
153             Put put = new Put(row1);
154             put.add(family, qualifier, data1);
155             table.put(tx1, put);
156 
157             assertTrue(tm.commitTableClient.tryInvalidateTransaction(tx1.getStartTimestamp()).get());
158             optionalCT = tm.commitTableClient.getCommitTimestamp(tx1.getStartTimestamp()).get();
159             assertTrue(optionalCT.isPresent());
160             CommitTimestamp ct = optionalCT.get();
161             assertFalse(ct.isValid());
162             assertEquals(CommitTable.INVALID_TRANSACTION_MARKER, ct.getValue());
163             assertTrue(ct.getLocation().compareTo(COMMIT_TABLE) == 0);
164 
165             // Finally test that we get the right commit timestamp for a committed tx
166             // that couldn't get
167             HBaseTransaction tx2 = (HBaseTransaction) tm.begin();
168             Put otherPut = new Put(row1);
169             otherPut.add(family, qualifier, data1);
170             table.put(tx2, otherPut);
171             try {
172                 tm.commit(tx2);
173             } catch (Exception e) { // (*) crash
174                 // Do nothing
175             }
176 
177             optionalCT = tm.commitTableClient.getCommitTimestamp(tx2.getStartTimestamp()).get();
178             assertTrue(optionalCT.isPresent());
179             ct = optionalCT.get();
180             assertTrue(ct.isValid());
181             assertEquals(tx2.getCommitTimestamp(), ct.getValue());
182             assertTrue(ct.getLocation().compareTo(COMMIT_TABLE) == 0);
183         }
184     }

188 
189         final long NON_EXISTING_CELL_TS = 1L;
190 
191         HBaseTransactionManager tm = (HBaseTransactionManager) newTransactionManager(context);
192 
193         try (TTable table = new TTable(hbaseConf, TEST_TABLE)) {
194 
195             // Test first we can not found a non-existent cell ts
196             HBaseCellId hBaseCellId = new HBaseCellId(table.getHTable(), row1, family, qualifier, NON_EXISTING_CELL_TS);
197             // Set an empty cache to allow to bypass the checking
198             CommitTimestampLocator ctLocator = new CommitTimestampLocatorImpl(hBaseCellId,
199                     Maps.<Long, Long>newHashMap());
200             Optional<CommitTimestamp> optionalCT = tm
201                     .readCommitTimestampFromShadowCell(NON_EXISTING_CELL_TS, ctLocator);
202             assertFalse(optionalCT.isPresent());
203 
204             // Then test that for a transaction committed, we get the right CT
205             HBaseTransaction tx1 = (HBaseTransaction) tm.begin();
206             Put put = new Put(row1);
207             put.add(family, qualifier, data1);
208             table.put(tx1, put);
209             tm.commit(tx1);
210             // Upon commit, the commit data should be in the shadow cells, so test it
211             optionalCT = tm.readCommitTimestampFromShadowCell(tx1.getStartTimestamp(), ctLocator);
212             assertTrue(optionalCT.isPresent());
213             CommitTimestamp ct = optionalCT.get();
214             assertTrue(ct.isValid());
215             assertEquals(tx1.getCommitTimestamp(), ct.getValue());
216             assertTrue(ct.getLocation().compareTo(SHADOW_CELL) == 0);
