145 
146         // Program the TSO to return an ad-hoc Timestamp and an commit response with heuristic actions
147         tso.queueResponse(new ProgrammableTSOServer.TimestampResponse(TX1_ST));
148         tso.queueResponse(new ProgrammableTSOServer.CommitResponse(true, TX1_ST, TX1_CT));
149         // Simulate that tx1 was committed by writing to commit table
150         commitTableWriter.addCommittedTransaction(TX1_ST, TX1_CT);
151         commitTableWriter.flush();
152         assertEquals(commitTable.countElements(), 1, "Rows should be 1!");
153 
154         try (TTable txTable = new TTable(hbaseConf, TEST_TABLE)) {
155             HBaseTransaction tx1 = (HBaseTransaction) tm.begin();
156             assertEquals(tx1.getStartTimestamp(), TX1_ST);
157             Put put = new Put(row1);
158             put.add(TEST_FAMILY.getBytes(), qualifier, data1);
159             txTable.put(tx1, put);
160             // Should succeed
161             tm.commit(tx1);
162 
163             // Check transaction status
164             assertEquals(tx1.getStatus(), Status.COMMITTED);
165             assertEquals(tx1.getCommitTimestamp(), TX1_CT);
166             // Check the cleanup process did its job and the committed data is there
167             // Note that now we do not clean up the commit table when exercising the heuristic actions
168             assertEquals(commitTable.countElements(), 1,
169                     "Rows should be 1! We don't have to clean CT in this case");
170             Optional<CommitTimestamp>
171                     optionalCT =
172                     tm.commitTableClient.getCommitTimestamp(TX1_ST).get();
173             assertTrue(optionalCT.isPresent());

237     public void testClientSuccessfullyCommitsWhenReceivingNotificationOfANewTSOAandCANTInvalidateTransaction()
238             throws Exception {
239 
240         // Program the TSO to return an ad-hoc Timestamp and a commit response with heuristic actions
241         tso.queueResponse(new ProgrammableTSOServer.TimestampResponse(TX1_ST));
242         tso.queueResponse(new ProgrammableTSOServer.CommitResponse(true, TX1_ST, TX1_CT));
243 
244         // Simulate that the original TSO was able to add the tx to commit table in the meantime
245         commitTableWriter.addCommittedTransaction(TX1_ST, TX1_CT);
246         commitTableWriter.flush();
247         assertEquals(commitTable.countElements(), 1, "Rows should be 1!");
248         SettableFuture<Optional<CommitTimestamp>> f1 = SettableFuture.create();
249         f1.set(Optional.<CommitTimestamp>absent());
250         SettableFuture<Optional<CommitTimestamp>> f2 = SettableFuture.create();
251         f2.set(Optional.of(new CommitTimestamp(COMMIT_TABLE, TX1_CT, true)));
252         doReturn(f1).doReturn(f2).when(commitTableClient).getCommitTimestamp(TX1_ST);
253 
254         try (TTable txTable = new TTable(hbaseConf, TEST_TABLE)) {
255             HBaseTransaction tx1 = (HBaseTransaction) tm.begin();
256             assertEquals(tx1.getStartTimestamp(), TX1_ST);
257             Put put = new Put(row1);
258             put.add(TEST_FAMILY.getBytes(), qualifier, data1);
259             txTable.put(tx1, put);
260 
261             tm.commit(tx1);
262 
263             // Check transaction status
264             assertEquals(tx1.getStatus(), Status.COMMITTED);
265             assertEquals(tx1.getCommitTimestamp(), TX1_CT);
266             // Check the cleanup process did its job and the committed data is there
267             // Note that now we do not clean up the commit table when exercising the heuristic actions
268             assertEquals(commitTable.countElements(), 1,
269                     "Rows should be 1! We don't have to clean CT in this case");
270             checkOperationSuccessOnCell(KeyValue.Type.Put, data1, TEST_TABLE.getBytes(), row1, TEST_FAMILY.getBytes(),
271                     qualifier);

315     protected void checkOperationSuccessOnCell(KeyValue.Type targetOp,
316                                                @Nullable byte[] expectedValue,
317                                                byte[] tableName,
318                                                byte[] row,
319                                                byte[] fam,
320                                                byte[] col) {
321 
322         try (HTable table = new HTable(hbaseConf, tableName)) {
323             Get get = new Get(row).setMaxVersions(1);
324             Result result = table.get(get);
325             Cell latestCell = result.getColumnLatestCell(fam, col);
326 
327             switch (targetOp) {
328                 case Put:
329                     assertEquals(latestCell.getTypeByte(), targetOp.getCode());
330                     assertEquals(CellUtil.cloneValue(latestCell), expectedValue);
331                     LOG.trace("Value for " + Bytes.toString(tableName) + ":"
332                             + Bytes.toString(row) + ":" + Bytes.toString(fam) + ":"
333                             + Bytes.toString(col) + "=>" + Bytes.toString(CellUtil.cloneValue(latestCell))
334                             + " (" + Bytes.toString(expectedValue) + " expected)");
335                     break;
336                 case Delete:
337                     LOG.trace("Value for " + Bytes.toString(tableName) + ":"
338                             + Bytes.toString(row) + ":" + Bytes.toString(fam)
339                             + Bytes.toString(col) + " deleted");
340                     assertNull(latestCell);
341                     break;
342                 default:
343                     fail();
344             }
345         } catch (IOException e) {
346             LOG.error("Error reading row " + Bytes.toString(tableName) + ":"
347                     + Bytes.toString(row) + ":" + Bytes.toString(fam)
348                     + Bytes.toString(col), e);
349             fail();
350         }
351     }
