32  @Test(groups = "sharedHBase")
33  public class TestSingleColumnFamily extends OmidTestBase {
34      private static final Logger LOG = LoggerFactory.getLogger(TestSingleColumnFamily.class);
35  
36  
37      @Test
38      public void testSingleColumnFamily(ITestContext context) throws Exception {
39          TransactionManager tm = newTransactionManager(context);
40          TTable table1 = new TTable(hbaseConf, TEST_TABLE);
41          int num = 10;
42          Transaction t = tm.begin();
43          for (int j = 0; j < num; j++) {
44              byte[] data = Bytes.toBytes(j);
45              Put put = new Put(data);
46              put.add(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value1"), data);
47              put.add(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value2"), data);
48              table1.put(t, put);
49          }
50          //tm.tryCommit(t);
51          //t=tm.beginTransaction(); //Visible if in a different transaction
52          Scan s = new Scan();
53          ResultScanner res = table1.getScanner(t, s);
54          Result rr;
55          int count = 0;
56          while ((rr = res.next()) != null) {
57              int tmp1 = Bytes.toInt(rr.getValue(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value1")));
58              int tmp2 = Bytes.toInt(rr.getValue(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value2")));
59              LOG.info("RES:" + tmp1 + ";" + tmp2);
60              count++;
61          }
62          assertTrue("Can't see puts. I should see "
63                          + num + " but I see " + count
64                  , num == count);
65  
66          tm.commit(t);
67          t = tm.begin();
68  
69          for (int j = 0; j < num / 2; j++) {
70              byte[] data = Bytes.toBytes(j);
71              byte[] ndata = Bytes.toBytes(j * 10);
72              Put put = new Put(data);
73              put.add(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value2"), ndata);
74              table1.put(t, put);
75          }
76          tm.commit(t);
77          t = tm.begin();
78          s = new Scan();
79          res = table1.getScanner(t, s);
80          count = 0;
81          int modified = 0, notmodified = 0;
82          while ((rr = res.next()) != null) {
83              int tmp1 = Bytes.toInt(rr.getValue(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value1")));
84              int tmp2 = Bytes.toInt(rr.getValue(Bytes.toBytes(TEST_FAMILY), Bytes.toBytes("value2")));
85              LOG.info("RES:" + tmp1 + ";" + tmp2);
86              count++;
87  
88              if (tmp2 == Bytes.toInt(rr.getRow()) * 10) {
89                  modified++;
90              } else {
91                  notmodified++;
92              }
93              if (count == 8) {
94                  LOG.debug("stop");
95              }
96          }
97          assertTrue("Can't see puts. I should see "
98                          + num + " but I see " + count
99                  , num == count);
100         assertTrue("Half of rows should equal row id, half not ("
101                         + modified + ", " + notmodified + ")"
102                 , modified == notmodified && notmodified == (num / 2));
103 
104         tm.commit(t);
105         LOG.info("End commiting");
106         table1.close();
107     }
