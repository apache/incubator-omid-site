103     @Test(timeOut = 10_000)
104     public void testBadFormedPackageThrowsException() throws Exception {
105 
106         // We need an instance throwing exceptions for this test
107         replyProcessor = spy(new ReplyProcessorImpl(new BlockingWaitStrategy(), metrics, new RuntimeExceptionPanicker(), batchPool));
108 
109         // Prepare test batch
110         Batch batch = batchPool.borrowObject();
111         batch.addCommitRetry(FIRST_ST, null, monCtx);
112         ReplyBatchEvent e = ReplyBatchEvent.EVENT_FACTORY.newInstance();
113         ReplyBatchEvent.makeReplyBatch(e, batch, 0);
114 
115         assertEquals(replyProcessor.nextIDToHandle.get(), 0);
116         assertEquals(replyProcessor.futureEvents.size(), 0);
117         assertEquals(batchPool.getNumActive(), 1);
118         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE - 1);
119 
120         try {
121             replyProcessor.onEvent(e, ANY_DISRUPTOR_SEQUENCE, false);
122             fail();
123         } catch (RuntimeException re) {
124             // Expected
125         }
126 
127         assertEquals(replyProcessor.nextIDToHandle.get(), 0);
128         assertEquals(replyProcessor.futureEvents.size(), 0);
129         assertEquals(batchPool.getNumActive(), 1);
130         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE - 1);

184     @Test(timeOut = 10_000)
185     public void testUnorderedSequenceOfBatchReplyEventsThatMustBeOrderedBeforeSendingReplies() throws Exception {
186 
187         // Prepare 3 batches with events and simulate a different order of arrival using the batch sequence
188 
189         // Prepare first a delayed batch (Batch #3)
190         Batch thirdBatch = batchPool.borrowObject();
191         thirdBatch.addTimestamp(FIRST_ST, mock(Channel.class), monCtx);
192         thirdBatch.addCommit(SECOND_ST, SECOND_CT, mock(Channel.class), monCtx);
193         ReplyBatchEvent thirdBatchEvent = ReplyBatchEvent.EVENT_FACTORY.newInstance();
194         ReplyBatchEvent.makeReplyBatch(thirdBatchEvent, thirdBatch, 2); // Set a higher sequence than the initial one
195 
196         assertEquals(replyProcessor.nextIDToHandle.get(), 0);
197         assertEquals(replyProcessor.futureEvents.size(), 0);
198         assertEquals(batchPool.getNumActive(), 1);
199         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE - 1);
200 
201         replyProcessor.onEvent(thirdBatchEvent, ANY_DISRUPTOR_SEQUENCE, false);
202 
203         assertEquals(replyProcessor.nextIDToHandle.get(), 0);
204         assertEquals(replyProcessor.futureEvents.size(), 1);
205         assertEquals(batchPool.getNumActive(), 1);
206         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE - 1);
207         assertFalse(thirdBatch.isEmpty());
208         verify(replyProcessor, never()).handleReplyBatchEvent(eq(thirdBatchEvent));
209 
210         // Prepare another delayed batch (Batch #2)
211         Batch secondBatch = batchPool.borrowObject();
212         secondBatch.addTimestamp(THIRD_ST, mock(Channel.class), monCtx);
213         secondBatch.addCommit(FOURTH_ST, FOURTH_CT, mock(Channel.class), monCtx);
214         ReplyBatchEvent secondBatchEvent = ReplyBatchEvent.EVENT_FACTORY.newInstance();
215         ReplyBatchEvent.makeReplyBatch(secondBatchEvent, secondBatch, 1); // Set another higher sequence
216 
217         replyProcessor.onEvent(secondBatchEvent, ANY_DISRUPTOR_SEQUENCE, false);
218 
219         assertEquals(replyProcessor.nextIDToHandle.get(), 0);
220         assertEquals(replyProcessor.futureEvents.size(), 2);
221         assertEquals(batchPool.getNumActive(), 2);
222         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE - 2);
223         assertFalse(secondBatch.isEmpty());
224         assertFalse(thirdBatch.isEmpty());
225 
226         // Finally, prepare the batch that should trigger the execution of the other two
227         Batch firstBatch = batchPool.borrowObject();
228         firstBatch.addAbort(FIFTH_ST, mock(Channel.class), monCtx);
229         ReplyBatchEvent firstBatchEvent = ReplyBatchEvent.EVENT_FACTORY.newInstance();
230         ReplyBatchEvent.makeReplyBatch(firstBatchEvent, firstBatch, 0); // Set the first batch with a higher sequence
231 
232         replyProcessor.onEvent(firstBatchEvent, ANY_DISRUPTOR_SEQUENCE, false);
233 
234         assertEquals(replyProcessor.nextIDToHandle.get(), 3);
235         assertEquals(replyProcessor.futureEvents.size(), 0);
236         assertEquals(batchPool.getNumActive(), 0);
237         assertEquals(batchPool.getNumIdle(), BATCH_POOL_SIZE);
238         assertTrue(firstBatch.isEmpty());
239         assertTrue(secondBatch.isEmpty());
240         assertTrue(thirdBatch.isEmpty());
241 
242         // Check the method calls have been properly ordered
243 
244         InOrder inOrderReplyBatchEvents = inOrder(replyProcessor, replyProcessor, replyProcessor);
245         inOrderReplyBatchEvents.verify(replyProcessor, times(1)).handleReplyBatchEvent(eq(firstBatchEvent));
246         inOrderReplyBatchEvents.verify(replyProcessor, times(1)).handleReplyBatchEvent(eq(secondBatchEvent));
247         inOrderReplyBatchEvents.verify(replyProcessor, times(1)).handleReplyBatchEvent(eq(thirdBatchEvent));
248 
249         InOrder inOrderReplies = inOrder(replyProcessor, replyProcessor, replyProcessor, replyProcessor, replyProcessor);
250         inOrderReplies.verify(replyProcessor, times(1)).sendAbortResponse(eq(FIFTH_ST), any(Channel.class));
251         inOrderReplies.verify(replyProcessor, times(1)).sendTimestampResponse(eq(THIRD_ST), any(Channel.class));
252         inOrderReplies.verify(replyProcessor, times(1)).sendCommitResponse(eq(FOURTH_ST), eq(FOURTH_CT), any(Channel.class));
253         inOrderReplies.verify(replyProcessor, times(1)).sendTimestampResponse(eq(FIRST_ST), any(Channel.class));
254         inOrderReplies.verify(replyProcessor, times(1)).sendCommitResponse(eq(SECOND_ST), eq(SECOND_CT), any(Channel.class));
255 
256     }
